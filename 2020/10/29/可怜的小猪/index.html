int poorPigs(int buckets, int minutesToDie, int minutesToTest) {
    if(buckets == 1)return 0;
    int matrixsize = minutesToTest / minutesToDie +1;
    int pigsnum = 0;
    int max = 0;
    while(max < buckets){
        pigsnum++;
        max = pow(matrixsize,pigsnum);
    }
    return pigsnum;
}
