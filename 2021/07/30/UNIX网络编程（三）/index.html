UNIX网络编程（三）

本文最后更新于：2 years ago

套接字编程

IPv4套接字地址结构

以sockaddr_in命名，定义在<netinet/in.h>头文件中：

struct in_addr{
    in_addr_t s_addr;
};

struct sockaddr_in{
    uint8_t			sin_len;		//地址结构长度字段
    sa_family_t		sin_family;		//地址族
    in_port_t		sin_port;		//端口号
    struct in_addr	sin_addr;		//IP地址
    char			sin_zero[8];	//无用
};

一些规范

  • sa_family_t 的类型可以是任意无符号整数
  • IPv4地址和TCP或UDP端口号在 socket地址结构中总是以网络字节序来存储。在后面将会区分主机字节序与网络字节序的区别。
  • IPv4地址存在两种不同的方法：
    • serv.sin_addr按照in_addr结构来引用地址
    • serv.sin_addr.s_addr按照in_addr_t来引用地址
    • 编译器对传递结构和传递整数的处理是完全不同的。需要注意
  • sin_zero未曾使用，但是总是把它置为0。按照惯例，我们通常在填写结构体之前，把整个结构体都置为0。使用bzero()函数
  • socket地址结构仅在给定主机上使用，虽然某些字段（IP地址和端口号）会用在通信中，但是结构本身是不会在主机上传递的。

通用套接字地址结构

在ANSI C之后，我们可以很轻松地传递地址结构的指针——使用void*。但是套接字结构是在ANSI C之前定义的，所以这也算是一个历史遗留问题。当时在<sys/socket.h>定义了一个通用的套接字地址结构：

struct sockaddr{
    uint8_t		sa_len;
    sa_family_t	sa_family;
    char		sa_data[14];
};

所以在使用套接字函数的时候，我们需要以一个指向通用套接字结构的指针作为参数，通常使用强制类型转换：

struct	sockaddr_in serv;
bind(sockfd, (struct sockaddr *)&serv, sizeof(serv));
bind(sockfd, (SA *)&serv, sizeof(serv));//本书中对该结构体宏定义SA

IPv6套接字地址结构

以sockaddr_in6命名，定义在<netinet/in.h>头文件中：

struct in6_addr{
    unit8_t		s6_addr[16];
};
#define SIN6_LEN
struct	sockaddr_in6{
    uint8_t			sin6_len;
    sa_family_t		sin6_family;
    in_port_t		sin6_port;
    uint32_t		sin6_flowinfo;
    struct in6_addr	sin6_addr;
    uint32_t		sin6_scope_id;
};

一些规范

  • 如果系统支持套接字地址结构中的长度字段，那么SIN6_LEN常值必须定义
  • IPv6的地址族是AF_INET6，而IPv4的地址族是AF_INET
  • sin6_flowinfo的使用仍然是一个研究课题
  • sin6_scope_id标识其范围，最常见的是链路局部地址的接口索引

新的通用套接字地址结构

struct sockaddr_storage{
    uint8_t		ss_len;
    sa_family_t	ss_family;
    ...
};

这个socket地址结构可以容纳系统所支持的任意套接字地质结构。

（1）如果系统支持的任意socket地址结构有对齐需要，那么sockaddr_storage能够满足最苛刻的对其要求

（2）sockaddr_storage足够大，能够容纳系统支持的任何socket地址结构。

（3）除了上面呈现的两个字段外，剩下的字段对于用户是透明地，需要类型强制转换后才能够访问到。

值-结果参数

1、从进程到内核传递的套接口结构有3个函数：bind，connect和sendto，这三个函数的一个参数是指向套接口结构的指针，另一个是结构的大小，如：

struct sockaddr_in serv;/*fill in serv[]*/
connect(sockfd,(SA *)&serv,sizeof(serv));

由于从进程到内核的参数是指向结构的指针和结构的大小，所以从进程到内核拷贝的数据是确定的。

2、与前面的正好相反，从内核到进程传递的套接口地址结构有四个函数：accept，recvfrom，getsockname，和getpeername，其参数是：指向套接字结构的指针和指向表示结构大小的指针，如：

struct sockaddr_in cli;
socklen_t len;

len = sizeof(cli);
getpeername(unixfd,(SA *)&cli,&len); /*len可能会发生改变*/

为何将结构大小由整数改为整数的指针呢？这是因为当函数被调用时，结构大小是一个值（value，此值告诉内核大小，使内核写结构时不至于越界），当函数返回时，结构大小又是一个结果（result，它告诉进程内核确实在结构中存储了信息），这种参数类型叫做值-结果参数。

字节排序函数

区分小端（little-endian）与大端（big-endian）

大端小端统称为主机字节序。比较遗憾的是，两种字节序都有主机使用。

网络协议必须指定一个网络字节序。因此我们要关注如何在主机字节序和网络字节序之间进行转换。下面是转换的四个函数：

#include <netinet/in.h>
uint16_t htons(uint16_t host16bitvalue);
uint32_t htonl(uint32_t host32bitvalue);

uint16_t ntohs(uint16_t net16bitvalue);
uint32_t ntohl(uint32_t nett32bitvalue);

readn函数：

循环调用read函数，从而读取n个字节

#include <errno.h>
#include <unistd.h>
//文件句柄/文件描述符是fd，int类型
ssize_t readn(int fd, void *buf, size_t n) {
    size_t nleft;	//剩余的字节数
    ssize_t nread;	//读取的字节数
    char *ptr;

    ptr = buf;	//指向缓冲区
    nleft = n;
    while (nleft > 0) {
        if ( (nread = read(fd, ptr, nleft)) < 0) {
            if (errno == EINTR) {
                nread = 0; /* call read() again */
            } else {
                return (-1);
            }
        } else if (nread == 0) {
            break;         /* EOF */
        }
        nleft -= nread;
        ptr += nread;
    }
    return (n - nleft);		/*return >= 0*/
}

writen函数：

往缓冲区中写n个字符

#include <unistd.h>
#include <errno.h>

ssize_t writen(int fd, const void *vptr, size_t n) {
    size_t nleft;
    ssize_t nwritten;
    const char *ptr;

    ptr = vptr;
    nleft = n;
    while (nleft > 0) {
        if ( (nwritten = write(fd, ptr, nleft)) <= 0) {
            if (nwritten < 0 && errno == EINTR) {
                nwritten = 0; /* call write() again */
            } else {
                return (-1); /* error */
            }
        } 
        nleft -= nwritten;
        ptr   += nwritten;
    }
    return (n - nwritten);
}

readline函数：

从缓冲区中读文本行，一次一个字节

#include <unistd.h>
#include <errno.h>

ssize_t readline(int fd, void *vptr, size_t maxlen) {
    ssize_t n, rc;
    char c, *ptr;

    ptr = vptr;
    for (n = 1; n <= maxlen; n++) {
        again:
        if ( (rc = read(fd, &c, 1)) == 1) {
            *ptr++ = c;
            if (c == '\n') {
                break;
            }
        } else if (rc == 0) {
            *ptr = 0;
            return (n - 1);
        } else {
            if (errno == EINTR) {
                goto again;
            } 
            return (-1);
        }
    }
    *ptr = 0;
    return ((maxlen == (n-1))?(n-1):n);
}

EINTR错误，表示系统调用被一个捕获的信号中断。发生该错误则继续读/写操作

上述的readline函数没读一个字节的数据，就要调用一次系统的read函数。这是非常低效的。改进版是自己使用一个内部缓冲区。

#include <errno.h>
#include <unistd.h>

#define MAXLINE 4096

static int read_cnt = 0;
static char *read_ptr;
static char read_buf[MAXLINE];

static ssize_t my_read(int fd, char *ptr) {
    if (read_cnt <= 0) {
        again:
        if ( (read_cnt = read(fd, read_buf, sizeof(read_buf))) < 0) {
            if (errno == EINTR) {
                goto again;
            }
            return (-1);            
        } else if (read_cnt == 0) {
            return (0);
        }
        read_ptr = read_buf;
    }
    read_cnt--;
    *ptr = *read_ptr++;
    return (1);
}

ssize_t readline(int fd, void *vptr, size_t maxlen) {
    ssize_t n, rc;
    char c, *ptr;

    ptr = vptr;
    for (n = 1; n < maxlen; n++) {
        if ( (rc = my_read(fd, &c)) == 1) {
            *ptr++ = c;
            if (c == '\n') {
                break;
            } 
        } else if (c == 0) {
            *ptr = 0;
            return (n - 1);
        } else {
            return (-1);
        }
    }
    *ptr = 0;
    return (n);
}

ssize_t readlinebuf(void **vptrptr) {
    if (read_cnt) {
        *vptrptr = read_ptr;
    }
    return (read_cnt);
}
