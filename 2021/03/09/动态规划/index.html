53. 最大子序和

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int pre = 0;
        int Max = nums[0];
        for(const auto &x : nums){
            pre = max(x+pre, x);
            Max = max(Max,pre);
        }
        return Max;
    }
};

152. 乘积最大子数组

由于是乘法，需要同时记录上一个状态的最大值、最小值

状态转移方程：

$f_{max}(i) = max_{1\leq i<n}{f_{max}(i−1)*a_i,f_{min}(i−1)×a_i,a_i}$

$f_{min}(i) = min_{1\leq i<n}{f_{max}(i−1)*a_i,f_{min}(i−1)×a_i,a_i}$

同时为了提高效率，需要在遍历过程中获取要返回的最值

代码实现

class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int ans = nums[0];
        int real_Max = nums[0];
        int real_Min = nums[0];
        for(int i = 1; i < nums.size(); i++){
            int Max = real_Max;
            int Min = real_Min;
            real_Max = max(Max*nums[i], max(Min*nums[i], nums[i]));
            real_Min = min(Max*nums[i], min(Min*nums[i], nums[i]));
            ans = max(real_Max,ans);
        }
        return ans;
    }
};

740. 删除与获得点数

算法思路

打家劫舍的进化版。

需要额外开辟一个cnt数组，来记录相同数字的出现次数就可以把它转换为经典的打家劫舍问题。

代码实现

class Solution {
public:
    int deleteAndEarn(vector<int>& nums) {
        int Max = 0;;
        for(int &x:nums)
            Max = max(x,Max);
        vector<int> cnt(Max+1,0), dp(Max+1,0);
        for(int &x:nums)
            cnt[x]++;
        dp[1] = cnt[1];
        for(int i = 2; i <= Max ; i++ ){
            dp[i] = max(dp[i-1], dp[i-2]+cnt[i]*i);
        }
        return dp[Max];
    }
};

代码实现：

class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        int m = text1.size();
        int n = text2.size();
        vector<vector<int>> dp(m+1, vector(n+1,0));
        for(int i = 1; i <= m; i ++){
            char text1_c = text1[i-1];
            for(int j  =1 ; j <= n; j ++){
                char text2_c = text2[j-1];
                if(text1_c == text2_c){
                    dp[i][j] = dp[i-1][j-1]+1;
                }else{
                    dp[i][j] = max(dp[i-1][j],dp[i][j-1]);
                }
            }
        }
        return dp[m][n];
    }
};
